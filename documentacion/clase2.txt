// ALGUNAS FUNCIONES QUE PARA OPERAR CON LAS LISTAS

// 1) includes
// es una funcion que devuelve true o false dependiendo si el PARAMETRO esta incluido en la lista
// ejemplo

var jugadoresDeFutbol = [ "juan", "pedro", "roberto" ];

jugadoresDeFutbol.includes("juan");

// ejemplo de un caso de error
var nuevaVariable = 3;
nuevaVariable.includes("juan"); // ME TIRA ERROR. PORQUE ES UN NUMEROOOOOOO, NO ES UN ARRAY

// TAMBIEN "includes" funciona para los string
var miString = "hola esto es un string";
miString.includes("hola");  // Esto devuelve TRUE - lo que recibe por parametro el includes para string es un substring

// 2) FILTER

// ejemplo

var jugadoresDeFutbol = [ "juan", "pedro", "roberto" ];
jugadoresDeFutbol.filter((jugador) => jugador.includes("o"))
